def countPrimes(n):
    if n < 2:
        return 0

    isPrime = [1] * n
    isPrime[0] = isPrime[1] = 0   # 0å’Œ1ä¸æ˜¯è´¨æ•°ï¼Œå…ˆæ’é™¤æ‰

    for i in range(2, int(n ** 0.5) + 1):

    return sum(isPrime)
